## Task 1: 创建 go_gallery.py

**核心逻辑：**

```python
def generate_gallery(output_path: Path) -> None:
    """生成输出画廊 HTML。"""
    from output_manager import list_runs
    
    runs = list_runs()
    html = _build_html(runs)
    output_path.write_text(html, encoding="utf-8")
    print(f"画廊已生成: {output_path}")
    webbrowser.open(output_path.absolute().as_uri())


def _build_html(runs: list[dict]) -> str:
    """构建自包含 HTML。"""
    cards_html = ""
    total_images = 0
    commands = set()
    
    for run in runs:
        cmd = run.get("command", "?")
        ts = (run.get("timestamp") or "?")[:19]
        rid = run.get("run_id", "?")
        images = run.get("images", [])
        params = run.get("params", {})
        total_images += len(images)
        commands.add(cmd)
        
        # 图片卡片
        imgs_div = ""
        run_dir = output_manager._get_output_dir() / rid
        for img_name in images:
            img_path = run_dir / "images" / img_name
            if img_path.is_file():
                imgs_div += f'<img src="file:///{img_path.as_posix()}" loading="lazy" />'
        
        # 参数摘要
        params_summary = ""
        for k, v in params.items():
            if k != "prompt_id" and k != "images":
                val = str(v)[:60]
                params_summary += f'<span class="tag">{k}: {val}</span>'
        
        prompt = params.get("prompt", "")
        if isinstance(prompt, str) and len(prompt) > 80:
            prompt = prompt[:80] + "..."
        
        cards_html += f"""
        <div class="card" data-command="{cmd}">
            <div class="images">{imgs_div}</div>
            <div class="meta">
                <span class="cmd-badge {cmd}">{cmd}</span>
                <span class="time">{ts}</span>
                <div class="prompt">{prompt}</div>
                <div class="params">{params_summary}</div>
            </div>
        </div>"""
    
    commands_str = " ".join(
        f'<button class="filter-btn" data-filter="{c}">{c}</button>'
        for c in sorted(commands)
    )
    
    return f"""<!DOCTYPE html>
<html lang="zh-CN">
<head>
<meta charset="UTF-8">
<meta name="viewport" content="width=device-width, initial-scale=1">
<title>AIGC Pipeline — Output Gallery</title>
<style>
  ...CSS...
</style>
</head>
<body>
<header>
  <h1>🎨 Output Gallery</h1>
  <div class="filters">
    <button class="filter-btn active" data-filter="all">All</button>
    {commands_str}
  </div>
</header>
<main id="gallery">
  {cards_html}
</main>
<footer>
  {len(runs)} runs · {total_images} images · generated {__import__('datetime').datetime.now().strftime('%Y-%m-%d %H:%M')}
</footer>
<script>
  document.querySelectorAll('.filter-btn').forEach(btn => {{
    btn.addEventListener('click', () => {{
      document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
      btn.classList.add('active');
      const filter = btn.dataset.filter;
      document.querySelectorAll('.card').forEach(card => {{
        card.style.display = filter === 'all' || card.dataset.command === filter ? '' : 'none';
      }});
    }});
  }});
</script>
</body>
</html>"""
```
