# V0.16.0 — 模型下载 实现方案

> **Goal:** `python -m agents model download <url>` — 从 HuggingFace / CivitAI / 直链下载模型到 ComfyUI 正确目录。
>
> **Architecture:** 新建 `agents/model_download.py`，复用 `model_manager.py` 的目录映射和模型根路径。
>
> **Tech Stack:** Python, requests (流式下载 + 进度条), `model_manager.CATEGORY_DIR_MAP`

---

## 用户交互

```
# 下载 LoRA（自动分类到 loras/）
python -m agents model download https://civitai.com/api/download/models/123456 --type lora

# 下载 checkpoint
python -m agents model download https://huggingface.co/stabilityai/.../sd_xl_base.safetensors --type checkpoint

# 指定文件名（可选）
python -m agents model download <url> --type lora --name my_lora.safetensors

# 使用 HF 镜像
python -m agents model download <hf_url> --type checkpoint --hf-mirror

# 仅预览（看会下载到什么目录）
python -m agents model download <url> --type lora --dry-run
```

## 类型 → 目录映射

| --type | 目标目录 |
|--------|----------|
| checkpoint | `checkpoints/` |
| lora | `loras/` |
| vae | `vae/` |
| clip | `clip/` 或 `text_encoders/` |
| embedding | `embeddings/` |
| controlnet | `controlnet/` |
| ipadapter | `ipadapter/` |

从 `model_manager.CATEGORY_DIR_MAP` 反向映射。

## Task 1: 创建 model_download.py

**核心函数:**

```python
import os, sys, json
from pathlib import Path
from typing import Optional
import requests

from model_manager import CATEGORY_DIR_MAP, resolve_models_root, _model_cache, _scan_models


def _resolve_target_dir(category: str) -> Optional[Path]:
    """根据类型确定目标子目录。"""
    models_root = resolve_models_root()
    if models_root is None:
        return None
    
    # 反向映射：category → 目录名
    cat_to_dir = {}
    for dirname, cat in CATEGORY_DIR_MAP.items():
        cat_to_dir.setdefault(cat, []).append(dirname)
    
    dirnames = cat_to_dir.get(category, [category + "s"])
    for d in dirnames:
        candidate = models_root / d
        if candidate.is_dir():
            return candidate
    # 取第一个候选目录
    return models_root / dirnames[0] if dirnames else models_root


def download_model(
    url: str,
    category: str = "checkpoint",
    *,
    filename: Optional[str] = None,
    hf_mirror: bool = False,
    civitai_token: Optional[str] = None,
    timeout: float = 600,
) -> Optional[Path]:
    """下载模型到 ComfyUI 对应目录。
    
    Returns: 下载后的文件路径，或 None 表示失败。
    """
    target_dir = _resolve_target_dir(category)
    if target_dir is None:
        print("错误: 无法确定 ComfyUI models 目录", file=sys.stderr)
        return None
    
    target_dir.mkdir(parents=True, exist_ok=True)
    
    # 处理 URL
    download_url = url
    headers = {}
    
    # CivitAI URL 处理
    if "civitai.com" in url.lower() and civitai_token:
        headers["Authorization"] = f"Bearer {civitai_token}"
    
    # HF mirror
    if hf_mirror and "huggingface.co" in url.lower():
        download_url = url.replace("huggingface.co", "hf-mirror.com")
    
    # 文件名：从 URL 推断或使用指定名
    if not filename:
        filename = url.split("/")[-1].split("?")[0]
        if not filename or filename.endswith("/"):
            filename = "model.safetensors"
    
    dest = target_dir / filename
    
    if dest.exists():
        size_mb = dest.stat().st_size / (1024 * 1024)
        print(f"⚠️  文件已存在: {dest} ({size_mb:.1f} MB)")
        print("   跳过下载，使用 --name 可保存为新文件名。")
        return dest
    
    # 流式下载
    print(f"下载: {filename}")
    print(f"目标: {dest}")
    print()
    
    try:
        r = requests.get(download_url, headers=headers, stream=True, timeout=timeout)
        r.raise_for_status()
    except requests.RequestException as e:
        print(f"下载失败: {e}", file=sys.stderr)
        return None
    
    total = int(r.headers.get("content-length", 0))
    downloaded = 0
    
    with open(dest, "wb") as f:
        for chunk in r.iter_content(chunk_size=8192):
            if chunk:
                f.write(chunk)
                downloaded += len(chunk)
                if total > 0:
                    pct = downloaded * 100 / total
                    mb = downloaded / (1024 * 1024)
                    print(f"\r  进度: {pct:.0f}% ({mb:.1f} MB)", end="", flush=True)
                else:
                    print(f"\r  已下载: {downloaded / (1024 * 1024):.1f} MB", end="", flush=True)
    
    print()
    print(f"\n✅ 下载完成: {dest} ({dest.stat().st_size / (1024 * 1024):.1f} MB)")
    
    # 刷新模型缓存
    global _model_cache
    _model_cache = _scan_models()
    
    return dest
```

**CLI main():**

```python
def main():
    import argparse
    
    parser = argparse.ArgumentParser(description="模型下载")
    parser.add_argument("url", help="下载 URL（HuggingFace / CivitAI / 直链）")
    parser.add_argument("--type", choices=["checkpoint", "lora", "vae", "clip",
        "embedding", "controlnet", "ipadapter", "upscale"],
        default="checkpoint", help="模型类型")
    parser.add_argument("--name", default=None, help="保存文件名（可选）")
    parser.add_argument("--hf-mirror", action="store_true", help="使用 HF 镜像")
    parser.add_argument("--civitai-token", default=None, help="CivitAI API Token")
    parser.add_argument("--dry-run", action="store_true", help="预览模式，不下载")
    args = parser.parse_args()
    
    if args.dry_run:
        target_dir = _resolve_target_dir(args.type)
        filename = args.name or args.url.split("/")[-1].split("?")[0] or "model.safetensors"
        print(f"\n预览: 下载 {args.type} → {target_dir / filename}")
        print(f"  URL: {args.url}")
        if "civitai.com" in args.url.lower():
            print(f"  需要 CivitAI Token: {'✅ 已提供' if args.civitai_token else '❌ 未提供'}")
        if "huggingface.co" in args.url.lower() and args.hf_mirror:
            print(f"  使用 HF 镜像: hf-mirror.com")
        return
    
    download_model(
        args.url,
        args.type,
        filename=args.name,
        hf_mirror=args.hf_mirror,
        civitai_token=args.civitai_token,
    )
```

**验证:**
```bash

# 输出预览信息，不实际下载
```

## Task 2: __main__.py — model download 挂到 models 子命令

`model download` 是 `models` 的子命令，需要修改 `_run_models()` 函数，加一个 `download` action。

**改动：**

```python
# 在 _run_models() 的 action == "check" 之后加:
    elif action == "download":
        from agents.model_download import main as download_main
        # 转发剩余参数
        sys.argv = [sys.argv[0], *sys.argv[3:]]
        download_main()
```

但这样不够干净。更简洁的做法：把 `model download` 注册为独立子命令（像 `caption`/`train` 一样），名字叫 `model-download`。但这样不符合 `models list/info/check` 的统一感。

更合理的做法：model_download.py 导出 `download_main()`，`_run_models()` 中调用时传入 args。

**实际实现：在 `_run_models()` 中加 `download` 分支：**

```python
    elif action == "download":
        if len(sys.argv) < 4:
            print("用法: python -m agents models download <url> [--type TYPE] [options]")
            return
        from agents.model_download import download_cli
        download_cli(sys.argv[3:])
```

`model_download.py` 导出 `download_cli(args_list)`:

```python
def download_cli(argv: list[str]) -> None:
    """从 argv 解析参数并执行下载。"""
    import argparse
    parser = argparse.ArgumentParser(prog="python -m agents models download")
    parser.add_argument("url", help="下载 URL")
    parser.add_argument("--type", ...)
    ...
    args = parser.parse_args(argv)
    if args.dry_run:
        ...
    else:
        download_model(...)
```
