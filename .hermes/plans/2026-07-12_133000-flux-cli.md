# V0.10.0 — Flux.2 Klein 原生 CLI 实现方案

> **Goal:** `python -m agents flux "prompt"` 专用 CLI，程序化构建 Flux 工作流，支持 9B/4B 权重和 LoRA 注入。
>
> **Architecture:** 新建 `agents/go_flux.py` 程序化构建 API 格式 workflow（不依赖 JSON 模板），注册到统一 CLI。
>
> **Reference:** comfyui-api skill 的 Flux.2 Klein 9B fp8 节点结构。

---

## 用户交互

```
python -m agents flux "赛博朋克少女，霓虹夜景，半身像"
python -m agents flux --raw "cinematic, cyberpunk girl, neon city, portrait"
python -m agents flux --lora knives_flux_lora.safetensors "白色连衣裙"
python -m agents flux --model 4b "prompt"              # 使用 4B 权重
python -m agents flux --width 1280 --height 720 "prompt"
```

## Flux.2 Klein 工作流节点图

```
UNETLoader(1) ──→ CFGGuider(10) ──→ SamplerCustomAdvanced(11) ──→ VAEDecode(12) ──→ SaveImage(13)
                        ↑
CLIPLoader(2) ──→ CLIPTextEncode(4) ──┬──→ CFGGuider(positive)
                                       └──→ ConditioningZeroOut(5) ──→ CFGGuider(negative)
VAELoader(3) ──────────────────────────────→ VAEDecode
EmptyFlux2LatentImage(6) ──────────────────→ SamplerCustomAdvanced
Flux2Scheduler(7) ─────────────────────────→ SamplerCustomAdvanced
KSamplerSelect(8) ─────────────────────────→ SamplerCustomAdvanced
RandomNoise(9) ────────────────────────────→ SamplerCustomAdvanced
```

**工作流构建代码（关键）：**

```python
def build_flux_workflow(
    prompt, negative_prompt="", seed=-1, steps=20, cfg=1.0,
    width=1024, height=1024, model_variant="9b",
    lora_name=None, lora_strength=1.0, sampler="euler", scheduler="normal",
):
    # 模型选择
    if model_variant == "9b":
        unet_name = "flux-2-klein-9b-fp8.safetensors"
        clip_name = "qwen_3_8b_fp8mixed.safetensors"
    else:
        unet_name = "flux-2-klein-4b-fp8.safetensors"
        clip_name = "qwen_3_06b_base.safetensors"
    vae_name = "flux2-vae.safetensors"
    
    # 节点计数器
    nid = [0]
    def next_id():
        nid[0] += 1
        return str(nid[0])
    
    wf = {}
    
    # 1. UNETLoader
    n1 = next_id()
    wf[n1] = {"class_type": "UNETLoader", "inputs": {
        "unet_name": unet_name, "weight_dtype": "default"}}
    
    # 2. CLIPLoader
    n2 = next_id()
    wf[n2] = {"class_type": "CLIPLoader", "inputs": {
        "clip_name": clip_name, "type": "flux2"}}
    
    # 3. VAELoader
    n3 = next_id()
    wf[n3] = {"class_type": "VAELoader", "inputs": {
        "vae_name": vae_name}}
    
    # 可选 LoRA 注入节点（在 UNET/CLIP 之后，其他节点之前）
    lora_model_output = [n1, 0]  # 默认连接 UNET 输出
    lora_clip_output = [n2, 0]   # 默认连接 CLIP 输出
    
    if lora_name:
        n_lora = next_id()
        wf[n_lora] = {"class_type": "LoraLoader", "inputs": {
            "model": [n1, 0], "clip": [n2, 0],
            "lora_name": lora_name,
            "strength_model": lora_strength,
            "strength_clip": lora_strength}}
        lora_model_output = [n_lora, 0]
        lora_clip_output = [n_lora, 1]
    
    # 4. CLIPTextEncode (positive)
    seed_actual = seed if seed != -1 else random.randint(1, 2**48 - 1)
    n4 = next_id()
    wf[n4] = {"class_type": "CLIPTextEncode", "inputs": {
        "text": prompt, "clip": lora_clip_output}}
    
    # 5. ConditioningZeroOut (negative)
    n5 = next_id()
    wf[n5] = {"class_type": "ConditioningZeroOut", "inputs": {
        "conditioning": [n4, 0]}}
    
    # 可选：使用用户提供的 negative prompt
    if negative_prompt:
        n_neg = next_id()
        wf[n_neg] = {"class_type": "CLIPTextEncode", "inputs": {
            "text": negative_prompt, "clip": lora_clip_output}}
        n5_wf = next_id()
        wf[n5_wf] = {"class_type": "ConditioningZeroOut", "inputs": {
            "conditioning": [n_neg, 0]}}
        negative = [n5_wf, 0]
    else:
        negative = [n5, 0]
    
    # 6. EmptyFlux2LatentImage
    n6 = next_id()
    wf[n6] = {"class_type": "EmptyFlux2LatentImage", "inputs": {
        "width": width, "height": height, "batch_size": 1}}
    
    # 7. Flux2Scheduler
    n7 = next_id()
    wf[n7] = {"class_type": "Flux2Scheduler", "inputs": {
        "steps": steps, "width": width, "height": height}}
    
    # 8. KSamplerSelect
    n8 = next_id()
    wf[n8] = {"class_type": "KSamplerSelect", "inputs": {
        "sampler_name": sampler}}
    
    # 9. RandomNoise
    n9 = next_id()
    wf[n9] = {"class_type": "RandomNoise", "inputs": {
        "noise_seed": seed_actual}}
    
    # 10. CFGGuider
    n10 = next_id()
    wf[n10] = {"class_type": "CFGGuider", "inputs": {
        "model": lora_model_output,
        "positive": [n4, 0],
        "negative": negative,
        "cfg": cfg}}
    
    # 11. SamplerCustomAdvanced
    n11 = next_id()
    wf[n11] = {"class_type": "SamplerCustomAdvanced", "inputs": {
        "noise": [n9, 0], "guider": [n10, 0],
        "sampler": [n8, 0], "sigmas": [n7, 0],
        "latent_image": [n6, 0]}}
    
    # 12. VAEDecode
    n12 = next_id()
    wf[n12] = {"class_type": "VAEDecode", "inputs": {
        "samples": [n11, 0], "vae": [n3, 0]}}
    
    # 13. SaveImage
    n13 = next_id()
    wf[n13] = {"class_type": "SaveImage", "inputs": {
        "images": [n12, 0], "filename_prefix": "flux_klein"}}
    
    return wf, seed_actual
```

**CLI 入口 main():**

```python
def main() -> None:
    parser = argparse.ArgumentParser(description="Flux.2 Klein 文生图")
    parser.add_argument("prompt", nargs="?", help="画面描述（自然语言，经 Ollama 转写）")
    parser.add_argument("--raw", action="store_true", help="跳过 Ollama，prompt 作正向提示词")
    parser.add_argument("--negative", default="", help="负向提示词")
    parser.add_argument("--seed", type=int, default=-1)
    parser.add_argument("--steps", type=int, default=20)
    parser.add_argument("--cfg", type=float, default=1.0)
    parser.add_argument("--width", type=int, default=1024)
    parser.add_argument("--height", type=int, default=1024)
    parser.add_argument("--model", choices=["9b", "4b"], default="9b", help="模型变体")
    parser.add_argument("--lora", default=None, help="LoRA 权重文件名")
    parser.add_argument("--lora-strength", type=float, default=1.0)
    parser.add_argument("--sampler", default="euler")
    parser.add_argument("--scheduler", default="normal")
    parser.add_argument("--count", type=int, default=1)
    args = parser.parse_args()
    
    # 处理 prompt
    user = args.prompt or input("请输入描述: ").strip()
    if not user:
        sys.exit(1)
    
    prompt = user if args.raw else optimize_prompt(user)
    
    wf, seed_actual = build_flux_workflow(
        prompt, negative_prompt=args.negative,
        seed=args.seed, steps=args.steps, cfg=args.cfg,
        width=args.width, height=args.height,
        model_variant=args.model,
        lora_name=args.lora, lora_strength=args.lora_strength,
        sampler=args.sampler, scheduler=args.scheduler,
    )
    
    # 提交
    result = comfy_post_prompt(wf)
    prompt_id = result.get("prompt_id")
    
    print(f"\nFlux.2 Klein ({args.model}) 已提交")
    print(f"  prompt_id: {prompt_id}")
    print(f"  seed: {seed_actual}")
    print(f"  正向: {prompt[:200]}...")
```
