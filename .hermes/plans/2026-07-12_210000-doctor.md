## Task 1: 创建 go_doctor.py

**核心逻辑：** 逐项检查，收集问题和建议。

```python
def run_doctor(auto_fix: bool = False) -> list[dict]:
    """执行全面诊断。返回检查结果列表。"""
    import sys, os, shutil, importlib
    from pathlib import Path
    
    checks = []
    
    def add_check(name, status, detail="", fix_hint=""):
        checks.append({
            "name": name, "status": status,
            "detail": detail, "fix_hint": fix_hint
        })
    
    # 1. Python
    v = sys.version_info
    add_check("Python 版本", v.major >= 3 and v.minor >= 10,
              f"Python {v.major}.{v.minor}.{v.micro}")
    
    # 2. ComfyUI
    from comfy_utils import check_comfy_health
    comfy_ok = check_comfy_health()
    add_check("ComfyUI", comfy_ok,
              "启动 ComfyUI 或检查 COMFY_URL 环境变量" if not comfy_ok else "")
    
    # 3. Ollama
    from comfy_utils import check_ollama_health
    ollama_ok = check_ollama_health()
    add_check("Ollama", ollama_ok,
              "启动 WSL Ollama (wsl sh -c 'ollama serve &')" if not ollama_ok else "")
    
    # 4. COMFY_ROOT
    from comfy_utils import resolve_comfy_root
    comfy_root = resolve_comfy_root()
    root_ok = comfy_root.is_dir() if comfy_root else False
    add_check("ComfyUI 目录", root_ok,
              str(comfy_root) if comfy_root else "设置 COMFY_ROOT 环境变量")
    
    # 5. models/
    from model_manager import resolve_models_root
    models_root = resolve_models_root()
    models_ok = models_root and models_root.is_dir()
    if not models_ok and auto_fix and models_root:
        models_root.mkdir(parents=True, exist_ok=True)
        models_ok = True
    add_check("models/ 目录", models_ok,
              str(models_root) if models_root else "ComfyUI 未安装",
              f"mkdir -p {models_root}" if models_root else "")
    
    # 6. 模型
    from model_manager import list_models
    all_models = list_models()
    has_models = len(all_models) > 0
    add_check("已安装模型", has_models,
              f"{len(all_models)} 个模型" if has_models else "下载模型到 models/ 目录")
    
    # 7. workflow
    from workflow_manager import list_workflows
    wfs = list_workflows()
    has_wfs = len(wfs) > 0
    add_check("Workflow 文件", has_wfs,
              f"{len(wfs)} 个" if has_wfs else "workflows/ 目录为空")
    
    # 8. pip 依赖
    deps_ok = all(importlib.util.find_slot(pkg) for pkg in ["requests"])
    deps_ok = True  # 简化
    try:
        import requests
        pip_ok = True
    except ImportError:
        pip_ok = False
    add_check("pip 依赖", pip_ok, "",
              "pip install -r requirements.txt" if not pip_ok else "")
    
    # 9. 磁盘
    try:
        usage = shutil.disk_usage(comfy_root or ".")
        free_gb = usage.free / (1024**3)
        disk_ok = free_gb > 5
        add_check("磁盘空间", disk_ok,
                  f"剩余 {free_gb:.0f} GB {'⚠️ 不足 5GB' if not disk_ok else ''}")
    except Exception:
        add_check("磁盘空间", True, "无法检查")
    
    return checks


def _print_report(checks: list[dict]) -> None:
    """打印诊断报告。"""
    ok = sum(1 for c in checks if c["status"])
    total = len(checks)
    
    print(f"\n  AIGC ComfyUI Pipeline — 诊断报告")
    print(f"  {'='*45}")
    print()
    
    for c in checks:
        icon = "✅" if c["status"] else "❌"
        detail = f" — {c['detail']}" if c["detail"] else ""
        print(f"  {icon} {c['name']}{detail}")
        if not c["status"] and c["fix_hint"]:
            print(f"     处理: {c['fix_hint']}")
    
    print(f"\n  {'='*45}")
    print(f"  {ok}/{total} 项通过")
    if ok < total:
        print(f"  {total - ok} 项需要处理")
    print(f"  {'='*45}\n")
```
