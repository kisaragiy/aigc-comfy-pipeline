# V0.21.0 — CLI 文档自动生成 实现方案

> **Goal:** 扫描 agents/ 所有 CLI 命令，自动生成 `docs/cli-reference.md` 完整参考文档。
>
> **Architecture:** 新建 `scripts/gen_cli_docs.py` 开发用脚本，遍历命令名 → 执行 `--help` → 写入 markdown。

> **Tech Stack:** Python, subprocess to capture --help output

## Task 1: 创建 gen_cli_docs.py

```python
"""自动生成 CLI 参考文档。"""
from __future__ import annotations

import subprocess
import sys
from pathlib import Path

ROOT = Path(__file__).resolve().parents[1]
DOCS = ROOT / "docs"
AGENTS = ROOT / "agents"

# 主命令列表（来自 __main__.py 的 _show_help）
COMMANDS = [
    ("run", "一句话提交 ComfyUI 文生图"),
    ("lora", "角色 LoRA 文生图"),
    ("ipa", "IPAdapter 锁脸"),
    ("multi", "多角色 LoRA 同图"),
    ("sweep", "参数网格扫描"),
    ("flux", "Flux.2 Klein 文生图"),
    ("workflow", "工作流模板管理"),
    ("models", "模型管理"),
    ("outputs", "产出管理"),
    ("queue", "ComfyUI 队列管理"),
    ("gallery", "输出画廊"),
    ("caption", "Ollama VL 自动标图"),
    ("train", "LoRA 训练编排"),
    ("check", "环境检查"),
    ("report", "管线验收报告"),
    ("doctor", "一键诊断修复"),
]

# 子命令（各内部 actions）
SUB_COMMANDS = {
    "workflow": ["list", "show <name>", "schema <name>", "check <name>", "convert <name>"],
    "models": ["list [category]", "info <name>", "check <workflow>", "download <url>"],
    "outputs": ["list", "show <id>", "clean"],
    "queue": ["list", "clear", "interrupt", "free"],
}


def run_help(command: str, sub: str | None = None) -> str:
    """运行命令并捕获 --help 输出。"""
    cmd = [sys.executable, "-m", "agents"]
    if sub:
        cmd.extend([command, sub])
    else:
        cmd.append(command)
    cmd.append("--help")

    try:
        r = subprocess.run(cmd, capture_output=True, text=True, timeout=30, cwd=ROOT)
        return r.stdout.strip() or r.stderr.strip()
    except subprocess.TimeoutExpired:
        return "(超时)"
    except Exception as e:
        return f"(错误: {e})"


def generate() -> str:
    lines = []
    lines.append("# CLI 参考文档")
    lines.append("")
    lines.append(f"> 自动生成于 {__import__('datetime').datetime.now().strftime('%Y-%m-%d %H:%M')}")
    lines.append("")
    lines.append("## 概述")
    lines.append("")
    lines.append(f"AIGC ComfyUI Pipeline v{__import__('agents').__version__}")
    lines.append("")

    for cmd_name, cmd_desc in COMMANDS:
        lines.append(f"---")
        lines.append("")
        lines.append(f"## `{cmd_name}`")
        lines.append("")
        lines.append(f"{cmd_desc}")
        lines.append("")
        lines.append("```")
        help_text = run_help(cmd_name)
        lines.append(help_text[:2000] if help_text else "(无帮助信息)")
        lines.append("```")
        lines.append("")

        # 子命令
        if cmd_name in SUB_COMMANDS:
            for sub in SUB_COMMANDS[cmd_name]:
                sub_label = sub.split()[0]
                lines.append(f"### `{cmd_name} {sub_label}`")
                lines.append("")
                lines.append("```")
                # 对需要参数的命令特殊处理
                if "<" in sub:
                    sub_text = run_help(cmd_name, sub_label)
                else:
                    sub_text = run_help(cmd_name, sub_label)
                lines.append(sub_text[:1500] if sub_text else "(无子命令帮助)")
                lines.append("```")
                lines.append("")

    return "\n".join(lines)


def main():
    content = generate()
    output_path = DOCS / "cli-reference.md"
    output_path.write_text(content, encoding="utf-8")
    print(f"✅ CLI 参考文档已生成: {output_path}")


if __name__ == "__main__":
    main()
```

**验证:**
```bash
python scripts/gen_cli_docs.py
# → 生成 docs/cli-reference.md
```
