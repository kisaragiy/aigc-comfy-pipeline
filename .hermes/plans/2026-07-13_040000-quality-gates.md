# V0.29.0 — 质量预设 + 自动门禁 实现方案

> **Goal:** 每个生成命令支持 `--preset` 优选参数 + 出图后自动验证+重试，大幅提高出图质量和稳定性。
>
> **Architecture:** 在 `comfy_utils.py` 加预设系统 + 质量门禁函数，各脚本加 3 个参数。

---

## Part A: 质量预设

```
--preset quality   (steps=40, cfg=7.0,  dpmpp_3m_sde, karras)  ← 高质量
--preset balanced  (steps=28, cfg=6.5,  dpmpp_2m,     karras)  ← 默认均衡
--preset fast      (steps=15, cfg=5.0,  euler,         normal)  ← 快速
--preset portrait  (width=896,height=1152, cfg=7.5)              ← 人像
```

环境变量 `AIGC_PRESET=quality` 可设全局默认。

## Part B: 自动质量门禁

```
--min-score 0.25   CLIP 评分低于此值自动重试（默认跳过）
--retry 3          最多重试次数
--no-validate      跳过验证（默认验证）
```

流程：
```
提交 → 出图 → validate → score >= min_score? → 保存
                          ↓ 否 & retry > 0
                        新 seed → 重试
```

## Task 1: comfy_utils.py — 预设 + 门禁

```python
# 质量预设
QUALITY_PRESETS = {
    "quality": {"steps": 40, "cfg": 7.0, "sampler_name": "dpmpp_3m_sde", "scheduler": "karras"},
    "balanced": {"steps": 28, "cfg": 6.5, "sampler_name": "dpmpp_2m", "scheduler": "karras"},
    "fast": {"steps": 15, "cfg": 5.0, "sampler_name": "euler", "scheduler": "normal"},
    "portrait": {"width": 896, "height": 1152, "cfg": 7.5, "sampler_name": "dpmpp_2m", "scheduler": "karras"},
}


def apply_preset(params: dict, preset: str | None = None) -> dict:
    """应用质量预设到参数。用户显式指定的参数优先。

    Args:
        params: 用户传入的参数 dict
        preset: 预设名（None 时从环境变量 AIGC_PRESET 读取）

    Returns:
        合并后的参数字典
    """
    if preset is None:
        preset = os.environ.get("AIGC_PRESET", "balanced")
    if preset not in QUALITY_PRESETS:
        print(f"[warn] 未知预设 '{preset}'，使用 balanced", file=sys.stderr)
        preset = "balanced"

    result = dict(QUALITY_PRESETS[preset])
    # 用户参数优先（只覆盖用户显式指定的）
    for k, v in params.items():
        if v is not None and k not in ("preset", "min_score", "retry", "no_validate"):
            result[k] = v
    print(f"[info] 质量预设: {preset} → steps={result.get('steps')}, cfg={result.get('cfg')}")
    return result


def generate_with_quality(
    build_fn: callable,
    prompt: str,
    *,
    min_score: float = 0.0,
    max_retries: int = 0,
    preset: str | None = None,
    **kwargs,
) -> dict:
    """生成 + 质量验证 + 自动重试。

    Args:
        build_fn: 工作流构建函数（如 build_flux_workflow）
        prompt: 提示词
        min_score: CLIP 评分阈值（≤0 跳过验证）
        max_retries: 最大重试次数
        preset: 质量预设名
        kwargs: 传给 build_fn 的参数

    Returns:
        {"workflow": wf, "seed": seed, "images": [...], "score": score, "retries": n}
    """
    import random

    # 应用预设
    params = apply_preset(kwargs, preset)
    params.pop("preset", None)

    has_validation = min_score > 0 and max_retries >= 0

    best_score = -1
    best_result = None

    for attempt in range(max_retries + 1):
        seed = params.get("seed", -1)
        if seed == -1 or attempt > 0:
            seed = random.randint(1, 2**48 - 1)

        wf, actual_seed = build_fn(prompt, seed=seed, **params)
        result = comfy_post_prompt(wf)
        pid = result.get("prompt_id", "")

        if pid == "dry-run":
            return {"workflow": wf, "seed": actual_seed, "images": [], "score": None, "retries": 0}

        base = comfy_base_url()
        images = wait_images(pid, base)

        image_paths = []
        comfy_root = resolve_comfy_root()
        for sub, name in images:
            path = (comfy_root / "output" / sub / name).resolve()
            if path.is_file():
                image_paths.append(str(path))

        score = None
        if has_validation and image_paths:
            try:
                from go_validate import validate_image
                v = validate_image(image_paths[0], prompt)
                clip = v.get("clip_score", {})
                if clip.get("available"):
                    score = clip["score"]
            except Exception:
                pass

        if score is not None and score > best_score:
            best_score = score
            best_result = (wf, actual_seed, image_paths, score)

        if score is not None and score >= min_score:
            return {"workflow": wf, "seed": actual_seed, "images": image_paths,
                    "score": score, "retries": attempt}

        if attempt < max_retries:
            print(f"  score={score:.3f} < {min_score}，重试 ({attempt+1}/{max_retries})")

    # 返回最佳结果
    if best_result:
        return {"workflow": best_result[0], "seed": best_result[1],
                "images": best_result[2], "score": best_result[3], "retries": max_retries}

    return {"workflow": wf, "seed": actual_seed, "images": image_paths,
            "score": score, "retries": attempt}
```

## Task 2-4: 各脚本加参数

每个脚本改动模式（以 `go_flux.py` 为例）：

```python
# 在 parser 加:
parser.add_argument("--preset", choices=["quality", "balanced", "fast", "portrait"],
                    default=None, help="质量预设")
parser.add_argument("--min-score", type=float, default=0.0,
                    help="最低 CLIP 评分（≤0 跳过验证）")
parser.add_argument("--retry", type=int, default=0,
                    help="质量不合格时最大重试次数")
parser.add_argument("--no-validate", action="store_true",
                    help="跳过质量验证")

# 调用处改为:
result = generate_with_quality(
    build_flux_workflow, prompt,
    min_score=args.min_score if not args.no_validate else 0.0,
    max_retries=args.retry,
    preset=args.preset,
    **params,
)
```
