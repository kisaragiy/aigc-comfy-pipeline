## 用户交互

```
# Prompt A vs B，同 seed 控制变量
python -m agents abtest --prompts "夕阳下的少女" "夜晚霓虹灯下的少女" --seed 42

# Best of N：同 prompt 多 seed 自动挑优（按 CLIP 评分排序）
python -m agents bestof "赛博朋克城市" --count 6

# 统一参数
python -m agents abtest --prompts "A" "B" --model 4b --lora xxx --dry-run
python -m agents bestof "prompt" --count 4 --steps 20 --raw
```

**CLI 入口:**

```python
def main():
    parser = argparse.ArgumentParser(description="A/B 测试 / Best of N")
    sub = parser.add_subparsers(dest="mode", required=True)
    
    # abtest
    p_ab = sub.add_parser("abtest")
    p_ab.add_argument("--prompts", nargs=2, required=True, help="A prompt vs B prompt")
    p_ab.add_argument("--seed", type=int, default=-1)
    p_ab.add_argument("--model", choices=["9b", "4b"], default="9b")
    p_ab.add_argument("--lora", default=None)
    p_ab.add_argument("--steps", type=int, default=20)
    p_ab.add_argument("--dry-run", action="store_true")
    
    # bestof
    p_bo = sub.add_parser("bestof")
    p_bo.add_argument("prompt", help="画面描述")
    p_bo.add_argument("--count", type=int, default=4)
    p_bo.add_argument("--model", choices=["9b", "4b"], default="9b")
    p_bo.add_argument("--lora", default=None)
    p_bo.add_argument("--steps", type=int, default=20)
    p_bo.add_argument("--dry-run", action="store_true")
    
    args = parser.parse_args()
```
