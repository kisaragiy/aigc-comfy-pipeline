### 核心逻辑

```python
def expand_grid(grid: dict) -> list[dict]:
    """展开网格参数为平面组合列表。
    
    输入: {"steps": [20, 30], "cfg": [1.0, 2.0]}
    输出: [{"steps": 20, "cfg": 1.0}, {"steps": 20, "cfg": 2.0}, ...]
    """
    if not grid:
        return [{}]
    keys = list(grid.keys())
    values = list(grid.values())
    from itertools import product
    combinations = []
    for combo in product(*values):
        combinations.append(dict(zip(keys, combo)))
    return combinations


def build_sweep_label(params: dict) -> str:
    """从参数组合生成文件标签。"""
    return "_".join(f"{k}{v}" for k, v in params.items())
```

### 网格执行 + 拼图

```python
def run_sweep(
    prompt: str,
    grid: dict,
    *,
    model_variant: str = "9b",
    lora_name: str | None = None,
    lora_strength: float = 1.0,
    negative: str = "",
    prefix: str = "sweep",
) -> None:
    from go_flux import build_flux_workflow
    from comfy_utils import comfy_post_prompt, comfy_base_url, wait_images, resolve_comfy_root
    from output_manager import save_run
    
    combinations = expand_grid(grid)
    print(f"网格扫描: {len(combinations)} 个组合")
    print(f"  {combinations}")
    
    results = []
    for i, params in enumerate(combinations):
        label = build_sweep_label(params)
        print(f"\n[{i+1}/{len(combinations)}] {label}...")
        
        steps = params.get("steps", 20)
        cfg_v = params.get("cfg", 1.0)
        seed_v = params.get("seed", -1)
        width_v = params.get("width", 1024)
        height_v = params.get("height", 1024)
        
        wf, seed_actual = build_flux_workflow(
            prompt=prompt,
            seed=seed_v, steps=steps, cfg=cfg_v,
            width=width_v, height=height_v,
            model_variant=model_variant,
            lora_name=lora_name, lora_strength=lora_strength,
            filename_prefix=f"{prefix}_{label}",
        )
        
        result = comfy_post_prompt(wf)
        pid = result.get("prompt_id", "")
        
        images = wait_images(pid, comfy_base_url())
        image_paths = []
        for sub, name in images:
            path = resolve_comfy_root() / "output" / sub / name
            if path.is_file():
                image_paths.append(str(path.resolve()))
        
        results.append({
            "params": params,
            "seed": seed_actual,
            "prompt_id": pid,
            "images": image_paths,
        })
    
    # 归档
    if results:
        all_images = [img for r in results for img in r["images"]]
        save_run("sweep-flux", all_images, {
            "prompt": prompt,
            "grid": grid,
            "model": model_variant,
            "lora": lora_name,
            "combinations": len(combinations),
        })
    
    # 生成对比拼图
    _make_grid(results, combinations, prefix)


def _make_grid(results, combinations, prefix):
    """生成带标注的对比拼图。"""
    try:
        from PIL import Image, ImageDraw, ImageFont
    except ImportError:
        print("[warn] Pillow 未安装，跳过对比拼图生成。pip install pillow")
        return
    
    # 收集所有有效图片
    valid = [(r, r["images"][0]) for r in results if r["images"]]
    if not valid:
        return
    
    n = len(valid)
    cols = min(4, n)
    rows = (n + cols - 1) // cols
    
    # 读取所有图片确定大小
    samples = [Image.open(im).convert("RGB") for _, im in valid]
    cell_w = max(im.width for im in samples)
    cell_h = max(im.height for im in samples)
    label_h = 40  # 顶部标注高度
    
    grid_img = Image.new("RGB", (cell_w * cols, (cell_h + label_h) * rows), (32, 32, 32))
    draw = ImageDraw.Draw(grid_img)
    
    for idx, (r, im_path) in enumerate(valid):
        col = idx % cols
        row = idx // cols
        x = col * cell_w
        y = row * (cell_h + label_h)
        
        # 标注文字
        label = ", ".join(f"{k}={v}" for k, v in r["params"].items())
        draw.rectangle([x, y, x + cell_w, y + label_h], fill=(48, 48, 48))
        try:
            font = ImageFont.truetype("arial.ttf", 14)
        except OSError:
            font = ImageFont.load_default()
        draw.text((x + 4, y + 4), label, fill=(255, 255, 255), font=font)
        
        # 图片
        im = Image.open(im_path).convert("RGB")
        im.thumbnail((cell_w, cell_h), Image.LANCZOS)
        paste_y = y + label_h
        grid_img.paste(im, (x + (cell_w - im.width) // 2, paste_y + (cell_h - im.height) // 2))
    
    grid_path = f"{prefix}_grid_comparison.jpg"
    grid_img.save(grid_path, quality=92)
    print(f"\n对比拼图已生成: {grid_path}")
```

### main() CLI

```python
def main() -> None:
    import argparse, json, sys
    
    parser = argparse.ArgumentParser(
        description="参数网格扫描 — Flux.2 Klein",
    )
    parser.add_argument("prompt", nargs="?", help="画面描述")
    parser.add_argument("--grid", required=True, help='JSON: {"steps":[20,30],"cfg":[1.0,2.0]}')
    parser.add_argument("--model", choices=["9b", "4b"], default="9b")
    parser.add_argument("--lora", default=None)
    parser.add_argument("--lora-strength", type=float, default=1.0)
    parser.add_argument("--negative", default="")
    parser.add_argument("--prefix", default="sweep")
    parser.add_argument("--raw", action="store_true", help="跳过 Ollama")
    args = parser.parse_args()
    
    user = args.prompt or input("请输入描述: ").strip()
    if not user:
        sys.exit(1)
    
    if args.raw:
        prompt = user
    else:
        from comfy_utils import optimize_prompt
        prompt = optimize_prompt(user)
    
    try:
        grid = json.loads(args.grid)
    except json.JSONDecodeError as e:
        print(f"网格参数 JSON 格式错误: {e}", file=sys.stderr)
        sys.exit(1)
    
    if not isinstance(grid, dict) or not grid:
        print("网格参数必须是非空 JSON 对象", file=sys.stderr)
        sys.exit(1)
    
    run_sweep(
        prompt,
        grid,
        model_variant=args.model,
        lora_name=args.lora,
        lora_strength=args.lora_strength,
        negative=args.negative,
        prefix=args.prefix,
    )
```
