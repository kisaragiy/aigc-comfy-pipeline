# V0.28.0 — API 服务化 实现方案

> **Goal:** `python -m agents serve` — 启动 REST API，将管线 CLI 暴露为 HTTP 接口。
>
> **Architecture:** FastAPI + uvicorn，异步作业队列。
>
> **Tech Stack:** FastAPI, uvicorn, Pydantic, Python asyncio

---

## API 设计

```
POST /api/flux          — Flux.2 Klein 文生图
POST /api/lora          — SDXL LoRA 文生图
POST /api/control       — ControlNet 引导生图
GET  /api/jobs/{id}     — 查询作业状态 + 结果
GET  /api/models        — 模型清单
GET  /api/workflows     — 工作流清单
GET  /api/health        — 健康检查
GET  /docs              — Swagger 交互文档
```

## 作业模型

```json
POST /api/flux
{
  "prompt": "赛博朋克少女",
  "steps": 20,
  "model": "9b"
}

→ 202 Accepted
{
  "job_id": "abc123",
  "status": "running"
}

GET /api/jobs/abc123
→ 200 OK
{
  "job_id": "abc123",
  "status": "completed",
  "result": {
    "image_url": "/outputs/xxx.png",
    "seed": 42,
    "params": {...}
  }
}
```

## Task 1: 创建 go_serve.py

**核心结构:**

```python
"""AIGC Pipeline REST API — FastAPI 服务。"""
from __future__ import annotations

import asyncio
import json
import os
import uuid
from pathlib import Path
from typing import Any

from fastapi import FastAPI, UploadFile, File, BackgroundTasks
from fastapi.responses import JSONResponse, FileResponse
from fastapi.staticfiles import StaticFiles
import uvicorn

app = FastAPI(title="AIGC Pipeline API", version="0.28.0")

# 作业存储（内存，生产环境可用 Redis）
jobs: dict[str, dict[str, Any]] = {}

# 静态文件服务（输出图片）
OUTPUTS_DIR = Path(__file__).resolve().parents[1] / "outputs"


def _run_job(job_id: str, command: str, params: dict) -> None:
    """后台执行出图作业。"""
    from comfy_utils import bootstrap_agents_path, comfy_base_url, comfy_post_prompt, wait_images, resolve_comfy_root
    
    jobs[job_id]["status"] = "running"
    try:
        if command == "flux":
            from go_flux import build_flux_workflow
            prompt = params.get("prompt", "")
            wf, seed = build_flux_workflow(prompt, **params)
            result = comfy_post_prompt(wf)
            pid = result.get("prompt_id", "")
            images = wait_images(pid, comfy_base_url())
            # 收集图片路径
            ...
        jobs[job_id]["status"] = "completed"
        jobs[job_id]["result"] = {"seed": seed, ...}
    except Exception as e:
        jobs[job_id]["status"] = "failed"
        jobs[job_id]["error"] = str(e)


@app.post("/api/flux")
async def api_flux(params: dict, background: BackgroundTasks):
    job_id = uuid.uuid4().hex[:12]
    jobs[job_id] = {"status": "queued", "command": "flux", "params": params}
    background.add_task(_run_job, job_id, "flux", params)
    return JSONResponse({"job_id": job_id, "status": "queued"}, status_code=202)


@app.get("/api/jobs/{job_id}")
async def get_job(job_id: str):
    job = jobs.get(job_id)
    if not job:
        return JSONResponse({"error": "not found"}, status_code=404)
    return job


@app.get("/api/health")
async def health():
    from comfy_utils import check_comfy_health, check_ollama_health
    return {
        "status": "ok",
        "comfyui": check_comfy_health(),
        "ollama": check_ollama_health(),
    }


@app.get("/api/models")
async def list_models():
    from model_manager import list_models
    return {"models": list_models()}


def main():
    port = int(os.environ.get("PORT", 8765))
    print(f"🚀 AIGC Pipeline API: http://127.0.0.1:{port}")
    print(f"📖 API 文档: http://127.0.0.1:{port}/docs")
    OUTPUTS_DIR.mkdir(parents=True, exist_ok=True)
    uvicorn.run(app, host="127.0.0.1", port=port)
```
