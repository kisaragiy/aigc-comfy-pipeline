## Task 1: 创建 go_report.py

**Objective:** 收集各模块数据，输出终端报告 + 可选 JSON

### 核心数据收集函数:

```python
import os, sys, json
from datetime import datetime
from pathlib import Path
from typing import Any

from agents import __version__


def _get_env_info() -> dict:
    return {
        "version": __version__,
        "python": sys.version.split()[0],
        "platform": sys.platform,
        "comfy_url": os.environ.get("COMFY_URL", "http://127.0.0.1:8188/prompt"),
        "ollama_url": os.environ.get("OLLAMA_URL", "http://127.0.0.1:11434/api/generate"),
    }


def _get_comfyui_info() -> dict:
    from comfy_utils import check_comfy_health, comfy_base_url
    base = comfy_base_url()
    info = {"online": check_comfy_health(), "base_url": base}
    if info["online"]:
        import requests
        try:
            r = requests.get(f"{base}/queue", timeout=5)
            queue = r.json() if r.status_code == 200 else {}
            info["queue_running"] = len(queue.get("queue_running", []))
            info["queue_pending"] = len(queue.get("queue_pending", []))
        except Exception:
            pass
        try:
            r = requests.get(f"{base}/system_stats", timeout=5)
            stats = r.json() if r.status_code == 200 else {}
            info["system"] = stats.get("system", {})
        except Exception:
            pass
    return info
```

### report() 函数:

```python
def generate_report(json_output: bool = False) -> dict:
    """生成完整管线报告。"""
    
    env = _get_env_info()
    comfy = _get_comfyui_info()
    ollama_ok = False
    ollama_models = []
    
    from comfy_utils import check_ollama_health
    ollama_ok = check_ollama_health()
    if ollama_ok:
        try:
            from comfy_utils import DEFAULT_OLLAMA_URL
            import requests
            base = DEFAULT_OLLAMA_URL.rstrip("/api/generate").rstrip("/")
            r = requests.get(f"{base}/api/tags", timeout=5)
            if r.status_code == 200:
                ollama_models = [m["name"] for m in r.json().get("models", [])]
        except Exception:
            pass
    
    # 模型清单
    models_total = 0
    models_by_cat = {}
    try:
        from model_manager import list_models
        all_models = list_models()
        models_total = len(all_models)
        for m in all_models:
            models_by_cat[m["category"]] = models_by_cat.get(m["category"], 0) + 1
    except Exception:
        pass
    
    # 工作流清单
    workflows_total = 0
    workflows_api = 0
    try:
        from workflow_manager import list_workflows
        wfs = list_workflows()
        workflows_total = len(wfs)
        workflows_api = sum(1 for w in wfs if w["is_api_format"])
    except Exception:
        pass
    
    # 产出统计
    recent_outputs = []
    outputs_total = 0
    try:
        from output_manager import list_runs
        runs = list_runs()
        outputs_total = len(runs)
        recent_outputs = [{
            "run_id": r.get("run_id", ""),
            "command": r.get("command", ""),
            "timestamp": r.get("timestamp", "")[:19] if r.get("timestamp") else "",
            "images": len(r.get("images", [])),
        } for r in runs[:5]]
    except Exception:
        pass
    
    data = {
        "report_time": datetime.now().isoformat()[:19],
        "environment": env,
        "comfyui": comfy,
        "ollama": {
            "online": ollama_ok,
            "models": ollama_models,
        },
        "models": {
            "total": models_total,
            "by_category": models_by_cat,
        },
        "workflows": {
            "total": workflows_total,
            "api_format": workflows_api,
            "ui_format": workflows_total - workflows_api,
        },
        "outputs": {
            "total": outputs_total,
            "recent": recent_outputs,
        },
        "ready": comfy.get("online", False) if comfy else False,
    }
    
    return data
```

### 终端显示函数:

```python
def _print_report(data: dict) -> None:
    env = data["environment"]
    comfy = data["comfyui"]
    ollama = data["ollama"]
    models = data["models"]
    wfs = data["workflows"]
    outs = data["outputs"]
    
    print(f"\n{'='*55}")
    print(f"  AIGC ComfyUI Pipeline — 验收报告")
    print(f"{'='*55}")
    print(f"  版本:    v{env['version']}")
    print()
    
    # 📋 环境信息
    print(f"📋 环境信息")
    print(f"  Python:    {env['python']}")
    print(f"  ComfyUI:   {env['comfy_url']}")
    print(f"  Ollama:    {env['ollama_url']}")
    print()
    
    # 🟢 ComfyUI
    c_icon = "✅ 在线" if comfy.get("online") else "❌ 离线"
    print(f"{'🟢' if comfy.get('online') else '🔴'} ComfyUI")
    print(f"  状态:   {c_icon}")
    if comfy.get("online"):
        qr = comfy.get("queue_running", 0)
        qp = comfy.get("queue_pending", 0)
        print(f"  队列:   {qr} 运行中, {qp} 待处理")
    print()
    
    # 🟡 Ollama
    o_icon = "✅ 在线" if ollama.get("online") else "❌ 离线"
    print(f"{'🟡' if ollama.get('online') else '🔴'} Ollama")
    print(f"  状态:   {o_icon}")
    if ollama.get("online") and ollama.get("models"):
        models_list = ", ".join(ollama["models"][:8])
        if len(ollama["models"]) > 8:
            models_list += f" ... (+{len(ollama['models'])-8})"
        print(f"  模型:   {models_list}")
    print()
    
    # 📦 模型
    print(f"📦 模型")
    print(f"  总计:   {models['total']}")
    for cat, count in sorted(models.get("by_category", {}).items()):
        print(f"    {cat}: {count}")
    print()
    
    # 📋 工作流
    print(f"📋 工作流")
    print(f"  总数:   {wfs['total']}")
    print(f"  API:    {wfs['api_format']}")
    print(f"  UI:     {wfs['ui_format']}")
    print()
    
    # 📊 产出
    print(f"📊 产出")
    print(f"  总运行: {outs['total']}")
    if outs.get("recent"):
        for r in outs["recent"]:
            print(f"    {r.get('timestamp',''):22s} {r.get('command',''):10s} {r.get('run_id','')}")
    print()
    
    # 总结
    ready = (
        comfy.get("online", False)
        and models["total"] > 0
        and wfs["api_format"] > 0
    )
    print(f"{'='*55}")
    print(f"  {'✅ 管线就绪' if ready else '⚠️  管线未完全就绪'}")
    if not comfy.get("online"):
        print(f"  处理: 启动 ComfyUI")
    if models["total"] == 0:
        print(f"  处理: 安装模型到 COMFY_ROOT/models/")
    if wfs["api_format"] == 0:
        print(f"  处理: workflow 需要 API 格式才能提交")
    print(f"{'='*55}\n")
```

### main():

```python
def main() -> None:
    import argparse
    parser = argparse.ArgumentParser(description="管线验收报告")
    parser.add_argument("--json", action="store_true", help="JSON 格式输出")
    args = parser.parse_args()
    
    data = generate_report(json_output=args.json)
    
    if args.json:
        print(json.dumps(data, ensure_ascii=False, indent=2))
    else:
        _print_report(data)
```
